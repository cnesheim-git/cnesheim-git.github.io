library(tidyverse)

event_medals <- read_csv("data/medals.csv")
medal_totals <- read_csv("data/medals_total.csv")

gdp_pop <- read_csv("data/gdp_pop.csv") |>
  rename (
    year = Time,
    country_name = `Country Name`,
    country_code = `Country Code`,
    gdp_current_usd = `GDP (current US$) [NY.GDP.MKTP.CD]`,
    gdp_per_capita = `GDP per capita (current US$) [NY.GDP.PCAP.CD]`,
    population = `Population, total [SP.POP.TOTL]`,
    gdp_growth_annpct = `GDP growth (annual %) [NY.GDP.MKTP.KD.ZG]`
  ) |>
  mutate(
    gdp_current_usd = na_if(gdp_current_usd, ".."),
    gdp_per_capita = na_if(gdp_per_capita, ".."),
    population = na_if(population, ".."),
    gdp_growth_annpct = na_if(gdp_growth_annpct, "..")
  ) |>
  select(year, country_code, gdp_current_usd, gdp_per_capita, population, gdp_growth_annpct) |>
  mutate(
    gdp_current_usd = as.numeric(gdp_current_usd),
    gdp_per_capita = as.numeric(gdp_per_capita),
    population = as.numeric(population),
    gdp_growth_annpct = as.numeric(gdp_growth_annpct)
  ) |>
  add_row(
    country_code = "TPE",
    gdp_current_usd = 797000000000,
    gdp_per_capita = 34060,
    population = 23400000,
    gdp_growth_annpct = 4.8
  ) |>
  add_row(
    country_code = "PRK",
    gdp_current_usd = NA,
    gdp_per_capita = NA,
    population = 26498823,
    gdp_growth_annpct = NA
  )

medal_gdp_pop_data <- left_join(medal_totals, gdp_pop, by="country_code") 

event_medals <- event_medals |>
  mutate(discipline = recode(discipline, "Athletics" = "Track & Field"))

medal_gdp_pop_scaled <- medal_gdp_pop_data |>
  mutate(
    tril_gdp = gdp_current_usd / 1000000000000,
    mil_pop = population / 1000000
  ) |>
  filter(!is.na(tril_gdp)) |>
  mutate(country = recode(country, "Türkiye" = "Turkey"))

medal_gdp_pop_filtered <- medal_gdp_pop_scaled |>
  filter(tril_gdp < 7) 

gdp_filtered_model <- lm(Total ~ tril_gdp, data=medal_gdp_pop_filtered)

medal_gdp_pop_filtered <- medal_gdp_pop_filtered |>
  mutate(resids = residuals(gdp_filtered_model), fitted = fitted(gdp_filtered_model)) |>
  arrange(desc(resids))

top3_countries <- medal_gdp_pop_filtered |> slice_max(order_by=resids, n=3)
bottom3_countries <- medal_gdp_pop_filtered |> slice_min(order_by=resids, n=3)
labeled_countries <- bind_rows(top3_countries, bottom3_countries)

medals_gdp_no_outliers <- ggplot() +
  geom_point(data=medal_gdp_pop_filtered, aes(x=tril_gdp, y=Total, size=mil_pop)) +

best_overperformers <- c("United States", "China", "Australia", "France", "Great Britain", "New Zealand", "Hungary", "Uzbekistan")

best_events <- event_medals |>
  filter(country %in% best_overperformers) |>
  group_by(country, discipline) |>
  mutate(dis_count = n()) |>
  ungroup() |>
  group_by(country) |>
  mutate(discipline = ifelse(dis_count < max(dis_count)*0.20, "Other", discipline)) |> 
  mutate(country_total = n()) |>
  ungroup() |>
  mutate(country = reorder(country, country_total)) |>
  mutate(discipline = factor(discipline, 
                             levels = c("Other", "Artistic Gymnastics", "Track & Field", "Swimming", "Diving", "Marathon Swimming", "Triathlon", "Cycling Track", "Cycling Mountain Bike", "Cycling BMX Racing", "Cycling Road", "Canoe Slalom", "Canoe Sprint", "Rowing", "Sailing", "Surfing", "Shooting", "Archery", "Equestrian", "Judo", "Fencing", "Taekwondo", "Boxing", "Weightlifting", "Wrestling", "Golf", "Basketball", "Rugby Sevens", "Badminton", "Table Tennis")))
